SELECT istprod_ministry_Staff.lastName, istprod_ministry_Staff.firstName, istprod_ministry_Staff.accountNo, istprod_ministry_ChangeRequest.requestdate, istprod_ministry_ChangeRequest.effectivedate, istprod_ministry_ChangeRequest.applieddate, istprod_ministry_ChangeRequest.type, istprod_ministry_ChangeRequest.fk_requestedBy, istprod_ministry_Authorization.role, istprod_ministry_Authorization.authorized, istprod_ministry_Authorization.sequence, istprod_ministry_Authorization.fk_AuthorizedBy, istprod_ministry_Authorization.authdate, istprod_ministry_FieldChange.field, istprod_ministry_FieldChange.oldValue, istprod_ministry_FieldChange.newValue, istprod_ministry_ChangeRequest.ChangeRequestID
FROM istprod_ministry_FieldChange INNER JOIN (istprod_ministry_Authorization INNER JOIN (istprod_ministry_Staff INNER JOIN (istprod_ministry_StaffChangeRequest INNER JOIN istprod_ministry_ChangeRequest ON istprod_ministry_StaffChangeRequest.ChangeRequestID = istprod_ministry_ChangeRequest.ChangeRequestID) ON istprod_ministry_Staff.accountNo = istprod_ministry_StaffChangeRequest.updateStaff) ON istprod_ministry_Authorization.fk_changeRequestID = istprod_ministry_ChangeRequest.ChangeRequestID) ON istprod_ministry_FieldChange.Fk_hasFieldChanges = istprod_ministry_ChangeRequest.ChangeRequestID
WHERE (((istprod_ministry_ChangeRequest.ChangeRequestID)="172016001080104122348391300505"))
ORDER BY istprod_ministry_Staff.lastName, istprod_ministry_Staff.firstName;
