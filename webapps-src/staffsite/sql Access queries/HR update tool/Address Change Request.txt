SELECT istprod_ministry_Staff.lastName, istprod_ministry_Staff.firstName, istprod_ministry_ChangeRequest.ChangeRequestID, istprod_ministry_Staff.accountNo, istprod_ministry_ChangeRequest.requestdate, istprod_ministry_ChangeRequest.effectivedate, istprod_ministry_ChangeRequest.applieddate, istprod_ministry_ChangeRequest.type, istprod_ministry_ChangeRequest.fk_requestedBy, istprod_ministry_FieldChange.field, istprod_ministry_FieldChange.oldValue, istprod_ministry_FieldChange.newValue
FROM (istprod_ministry_Staff INNER JOIN (istprod_ministry_StaffChangeRequest INNER JOIN istprod_ministry_ChangeRequest ON istprod_ministry_StaffChangeRequest.ChangeRequestID = istprod_ministry_ChangeRequest.ChangeRequestID) ON istprod_ministry_Staff.accountNo = istprod_ministry_StaffChangeRequest.updateStaff) INNER JOIN istprod_ministry_FieldChange ON istprod_ministry_ChangeRequest.ChangeRequestID = istprod_ministry_FieldChange.Fk_hasFieldChanges
WHERE (((istprod_ministry_Staff.lastName)="Al-Bahsous"))
ORDER BY istprod_ministry_Staff.lastName, istprod_ministry_Staff.firstName, istprod_ministry_ChangeRequest.ChangeRequestID;
