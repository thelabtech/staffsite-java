SELECT istprod_crs_CRSPerson.firstName, istprod_crs_CRSPerson.lastName, istprod_crs_CRSConference.name, istprod_crs_CRSPayment.paymentDate, istprod_crs_CRSPayment.debit, istprod_crs_CRSPayment.credit, istprod_crs_CRSPayment.type, istprod_crs_CRSPayment.businessUnit, istprod_crs_CRSPayment.dept, istprod_crs_CRSPayment.region, istprod_crs_CRSPayment.project, istprod_crs_CRSPayment.accountNo, istprod_crs_CRSPayment.comment
FROM (istprod_crs_CRSPayment INNER JOIN (istprod_crs_CRSPerson INNER JOIN istprod_crs_CRSRegistration ON istprod_crs_CRSPerson.personID = istprod_crs_CRSRegistration.fk_PersonID) ON istprod_crs_CRSPayment.fk_RegistrationID = istprod_crs_CRSRegistration.registrationID) INNER JOIN istprod_crs_CRSConference ON istprod_crs_CRSRegistration.fk_ConferenceID = istprod_crs_CRSConference.conferenceID;
