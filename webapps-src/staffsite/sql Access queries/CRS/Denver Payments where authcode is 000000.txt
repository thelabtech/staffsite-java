SELECT istprod_crs_CRSPerson.firstName, istprod_crs_CRSPerson.lastName, istprod_crs_CRSPerson.email, istprod_crs_CRSPayment.debit, istprod_crs_CRSPayment.postedDate, istprod_crs_CRSPayment.credit, istprod_crs_CRSPayment.paymentDate, istprod_crs_CRSPayment.type, istprod_crs_CRSPayment.authCode
FROM ((istprod_crs_CRSConference INNER JOIN istprod_crs_CRSRegistration ON istprod_crs_CRSConference.conferenceID = istprod_crs_CRSRegistration.fk_ConferenceID) INNER JOIN istprod_crs_CRSPayment ON istprod_crs_CRSRegistration.registrationID = istprod_crs_CRSPayment.fk_RegistrationID) INNER JOIN istprod_crs_CRSPerson ON istprod_crs_CRSRegistration.fk_PersonID = istprod_crs_CRSPerson.personID
WHERE (((istprod_crs_CRSPayment.authCode)="000000") AND ((istprod_crs_CRSConference.conferenceID)=22));
