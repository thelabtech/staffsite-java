FROM (istprod_crs_Person INNER JOIN istprod_simplesecuritymanager_user ON istprod_crs_Person.username = istprod_simplesecuritymanager_user.username) INNER JOIN (istprod_crs_Conference INNER JOIN istprod_crs_Registration ON istprod_crs_Conference.conferenceID = istprod_crs_Registration.fk_ConferenceID) ON istprod_crs_Person.personID = istprod_crs_Registration.fk_PersonID;
