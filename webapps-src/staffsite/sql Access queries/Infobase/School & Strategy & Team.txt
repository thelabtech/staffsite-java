SELECT istprod_ministry_TargetArea.name, istprod_ministry_Activity.strategy, istprod_ministry_Activity.status, istprod_ministry_Activity.periodBegin, istprod_ministry_Activity.periodEnd, istprod_ministry_TargetArea.TargetAreaID, istprod_ministry_Activity.ActivityID, istprod_ministry_LocalLevel.name, istprod_ministry_LocalLevel.teamID
FROM (istprod_ministry_Activity INNER JOIN istprod_ministry_TargetArea ON istprod_ministry_Activity.fk_targetAreaID = istprod_ministry_TargetArea.TargetAreaID) INNER JOIN istprod_ministry_LocalLevel ON istprod_ministry_Activity.fk_teamID = istprod_ministry_LocalLevel.teamID;
