SELECT istprod_ministry_TargetArea.name, istprod_ministry_Activity.ActivityID, istprod_ministry_Activity.status, istprod_ministry_Activity.periodBegin, istprod_ministry_Activity.periodEnd, istprod_ministry_Activity.strategy, istprod_ministry_LocalLevel.name, istprod_ministry_Staff.firstName, istprod_ministry_Staff.lastName
FROM (istprod_ministry_TargetArea INNER JOIN ((istprod_ministry_Activity INNER JOIN istprod_ministry_Assoc_activityContact ON istprod_ministry_Activity.ActivityID = istprod_ministry_Assoc_activityContact.ActivityID) INNER JOIN istprod_ministry_LocalLevel ON istprod_ministry_Activity.fk_teamID = istprod_ministry_LocalLevel.teamID) ON istprod_ministry_TargetArea.TargetAreaID = istprod_ministry_Activity.fk_targetAreaID) INNER JOIN istprod_ministry_Staff ON istprod_ministry_Assoc_activityContact.accountNo = istprod_ministry_Staff.accountNo;
