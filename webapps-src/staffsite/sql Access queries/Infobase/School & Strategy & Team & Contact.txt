SELECT istprod_ministry_TargetArea.name, istprod_ministry_TargetArea.city, istprod_ministry_TargetArea.state, istprod_ministry_LocalLevel.name, istprod_ministry_Staff.firstName, istprod_ministry_Staff.lastName
FROM (((istprod_ministry_Activity INNER JOIN istprod_ministry_TargetArea ON istprod_ministry_Activity.fk_targetAreaID = istprod_ministry_TargetArea.TargetAreaID) INNER JOIN istprod_ministry_LocalLevel ON istprod_ministry_Activity.fk_teamID = istprod_ministry_LocalLevel.teamID) INNER JOIN istprod_ministry_Assoc_activityContact ON istprod_ministry_Activity.ActivityID = istprod_ministry_Assoc_activityContact.ActivityID) INNER JOIN istprod_ministry_Staff ON istprod_ministry_Assoc_activityContact.accountNo = istprod_ministry_Staff.accountNo
WHERE (((istprod_ministry_Activity.strategy)="CA") AND ((istprod_ministry_Activity.status)="FR"));
